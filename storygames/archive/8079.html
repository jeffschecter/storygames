\n Hi guys, here's a macro to make playing IAWA online less awkward. This is designed for the beta of Maptool, which should be out of beta soon and is fairly stable (I haven't had a crash but I don't play with a lot of advanced features besides this macro).
\n
\n[H: fail = input("Dice1 | d12, d10, d8, d6, d4 | First Form | RADIO | ORIENT=H VALUE=STRING",
\n"Dice2 | d10, d8, d6, d4 | Second Form | RADIO | ORIENT=H VALUE=STRING",
\n"Dice3 | 0, d8, d10, d12 | Particular Strength | RADIO | ORIENT=H VALUE=STRING",
\n"Dice4 | 0, 1d6, 2d6 | Advantage + Conviction | RADIO | ORIENT=H VALUE=STRING")]
\n[H:abort(fail)]
\n[H:DiceList = ""]
\n[H,C(4): DiceList = DiceList + eval("Dice" + (roll.count+1)) + if((roll.count+1) < 4 ,"," ,"")]
\n[H:ResultList = ""]
\n[H,C(listCount(DiceList),''): ResultList = ResultList + eval(listGet(DiceList, roll.count)) + if((roll.count+1) < listCount(DiceList) ,"," ,"")]
\n [H:AddRolls = 0]
\n[H,C(1): AddRolls = AddRolls + eval(listGet(ResultList, 3))]
\n [H:HighestRoll = 0]
\n[H:SecondRoll = 0]
\n[H:LowestRoll = 2]
\n[H,C(3):HighestRoll = if(eval(listGet(ResultList, roll.count)) > eval(listGet(ResultList, HighestRoll)), roll.count, HighestRoll)]
\n[H,C(3):LowestRoll = if(eval(listGet(ResultList, roll.count)) < eval(listGet(ResultList, LowestRoll)), roll.count, LowestRoll)]
\n[H,C(3):SecondRoll = if(HighestRoll != roll.count && LowestRoll != roll.count, roll.count, SecondRoll)]
\n [eval(listGet(ResultList, HighestRoll)+ "+" + AddRolls)], [eval(listGet(ResultList, SecondRoll))], [eval(listGet(ResultList, LowestRoll))] ({DiceList})
\n
